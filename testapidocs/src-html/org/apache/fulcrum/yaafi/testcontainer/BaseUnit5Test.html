022import org.apache.avalon.framework.component.Component;
023import org.apache.avalon.framework.component.ComponentException;
024import org.apache.avalon.framework.service.ServiceException;
025import org.junit.jupiter.api.AfterEach;
026
027/**
028 * Base class for unit tests for components. This version doesn't load the
029 * container until the first request for a component. This allows the tester to
030 * populate the configurationFileName and roleFileName, possible one per test.
031 *
032 * @author <a href="mailto:epugh@upstate.com">Eric Pugh</a>
033 * @author <a href="mailto:quintonm@bellsouth.net">Quinton McCombs</a>
034 */
035public abstract class BaseUnit5Test {
036        
037        /** YaffiContainer for the components */
038        private Container container;
039        
040        /** Setup our default configurationFileName */
041        private String configurationFileName = "src/test/TestComponentConfig.xml";
042        
043        /** Setup our default roleFileName */
044        private String roleFileName = "src/test/TestRoleConfig.xml";
045        
046        /** Setup our paramterFileName */
047        private String parameterFileName = "src/test/TestParameters.properties";
048
049        /**
050         * Gets the configuration file name for the container should use for this test.
051         * By default it is src/test/TestComponentConfig.
052         *
053         * @param configurationFileName config file name
054         */
055        protected void setConfigurationFileName(String configurationFileName) {
056                this.configurationFileName = configurationFileName;
057        }
058
059        /**
060         * Override the role file name for the container should use for this test. By
061         * default it is src/test/TestRoleConfig.
062         *
063         * @param roleFileName role file name
064         */
065        protected void setRoleFileName(String roleFileName) {
066                this.roleFileName = roleFileName;
067        }
068
069        /**
070         * Override the parameter file name for the container should use for this test.
071         * By default it is src/test/TestRoleConfig.
072         *
073         * @param parameterFileName the name of the parameter file
074         */
075        protected void setParameterFileName(String parameterFileName) {
076                this.parameterFileName = parameterFileName;
077        }
078
079        @AfterEach
080        protected void tearDown() throws Exception {
081                if (this.container != null) {
082                        this.container.dispose();
083                }
084                this.container = null;
085        }
086
087        /**
088         * Gets the configuration file name for the container should use for this test.
089         *
090         * @return The filename of the configuration file
091         */
092        protected String getConfigurationFileName() {
093                return this.configurationFileName;
094        }
095
096        /**
097         * Gets the role file name for the container should use for this test.
098         *
099         * @return The filename of the role configuration file
100         */
101        protected String getRoleFileName() {
102                return this.roleFileName;
103        }
104
105        /**
106         * Gets the parameter file name for the container should use for this test.
107         *
108         * @return The filename of the parameter file
109         */
110        protected String getParameterFileName() {
111                return this.parameterFileName;
112        }
113
114        /**
115         * Returns an instance of the named component. Starts the container if it hasn't
116         * been started.
117         *
118         * @param roleName Name of the role the component fills.
119         * @return Object representing the named component
120         * @throws ComponentException if the component is not found
121         */
122        protected Object lookup(String roleName) throws ComponentException {
123                if (this.container == null) {
124                        this.container = new Container();
125                        this.container.startup(getConfigurationFileName(), getRoleFileName(), getParameterFileName());
126                }
127                return this.container.lookup(roleName);
128        }
129
130        /**
131         * Releases the component
132         *
133         * @param component the component to be released
134         */
135        protected void release(Component component) {
136                if (this.container != null) {
137                        this.container.release(component);
138                }
139        }
140
141        /**
142         * Releases the component
143         *
144         * @param component the component to be released
145         */
146        protected void release(Object component) {
147                if (this.container != null) {
148                        this.container.release(component);
149                }
150        }
151
152        /**
153         * Decommision the service
154         * 
155         * @param name the name of the service
156         * @throws ServiceException if the service is not found
157         */
158        protected void decommision(String name) throws ServiceException {
159                if (this.container != null) {
160                        this.container.decommission(name);
161                }
162        }
163}
