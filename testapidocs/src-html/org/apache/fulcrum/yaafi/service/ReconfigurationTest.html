029import junit.framework.TestCase;
030
031/**
032 * Test suite for the ReconfigurationService. This test doesn't do anything
033 * apart from running a minute so you have some time to tinker with the
034 * component configuration file.
035 *
036 * @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl</a>
037 */
038
039public class ReconfigurationTest extends TestCase {
040        private ServiceContainer container = null;
041
042        /**
043         * Constructor
044         * 
045         * @param name the name of the test case
046         */
047        public ReconfigurationTest(String name) {
048                super(name);
049        }
050
051        /*
052         * (non-Javadoc)
053         * 
054         * @see junit.framework.TestCase#tearDown()
055         */
056        protected void tearDown() throws Exception {
057                ServiceContainerFactory.dispose(this.container);
058                super.tearDown();
059        }
060
061        /**
062         * @return get our simple test component
063         * @throws ServiceException if service not found
064         */
065        private TestComponent getTestComponent() throws ServiceException {
066                return (TestComponent) container.lookup(TestComponent.ROLE);
067        }
068
069        /**
070         * Trigger the ReconfigurationService by instantiating it manually.
071         * 
072         * @throws Exception generic exception
073         */
074        public void testReconfigurationService() throws Exception {
075                // instantiate a YAAFI container
076
077                ServiceContainerConfiguration config = new ServiceContainerConfiguration();
078                config.loadContainerConfiguration("./src/test/TestYaafiContainerConfig.xml");
079                this.container = ServiceContainerFactory.create(config);
080
081                // the ReconfigurationService is configured to be instantiated on demand
082                // get an instance to start monitoring ...
083
084                ReconfigurationService reconfigurationService = null;
085
086                reconfigurationService = (ReconfigurationService) this.container.lookup(ReconfigurationService.class.getName());
087
088                assertNotNull(reconfigurationService);
089
090                // comment out if you want to tinker with componentConfiguration manually
091
092                // Thread.sleep(60000);
093
094                this.getTestComponent().test();
095        }
096}
