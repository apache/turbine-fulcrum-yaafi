033import org.apache.fulcrum.yaafi.service.advice.AdviceService;

038/**
039 * Test suite for the ServiceManagereService.
040 *
041 * @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl</a>
042 */
043
044public class AdviceServiceTest  implements DependentTestComponent
045{
046    private AdviceService service;
047    private DependentTestComponent advisedThis;
048    private ServiceContainer container;
049
050
051    @BeforeEach
052    protected void setUp() throws Exception
053    {
054        ServiceContainerConfiguration config = new ServiceContainerConfiguration();
055        config.loadContainerConfiguration( "./src/test/TestYaafiContainerConfig.xml" );
056        this.container = ServiceContainerFactory.create( config );
057        service = (AdviceService) this.container.lookup(AdviceService.class.getName());
058    }
059
060    @AfterEach
061    protected void tearDown() throws Exception
062    {
063        ServiceContainerFactory.dispose(this.container);
064    }
065
066    /**
067     *  Advice a StringBuilder based on the CharSequence interface
068     * @throws Exception generic exception
069     */
070    @Test
071    public void testSimpleAdvice() throws Exception
072    {
073        String[] interceptorList = { LoggingInterceptorService.class.getName() };
074        StringBuilder unadvicedObject = new StringBuilder("foo");
075        CharSequence advicedObject = (CharSequence) this.service.advice(unadvicedObject );
076
077        int length = advicedObject.length();
078        assertTrue(this.service.isAdviced(advicedObject));
079        assertFalse(this.service.isAdviced(unadvicedObject));
080        assertTrue(unadvicedObject.length() == length);
081    }
082
083    /**
084     * Advice a StringBuilder based on the CharSequence interface
085     * @throws Exception generic exception
086     */
087    @Test
088    public void testSimpleObject() throws Exception
089    {
090        String[] interceptorList = { LoggingInterceptorService.class.getName() };
091        StringBuilder unadvicedObject = new StringBuilder("foo");
092        CharSequence advicedObject = (CharSequence) this.service.advice("adviced", interceptorList, unadvicedObject );
093
094        int length = advicedObject.length();
095        assertTrue(this.service.isAdviced(advicedObject));
096        assertFalse(this.service.isAdviced(unadvicedObject));
097        assertTrue(unadvicedObject.length() == length);
098    }
099
100    /**
101     * Advice a StringBuilder based on the CharSequenceInterface with default interceptors
102     * @throws Exception generic exception
103     */
104    @Test
105    public void testDefaultAdvice() throws Exception
106    {
107        StringBuilder unadvicedObject = new StringBuilder("foo");
108        CharSequence advicedObject = (CharSequence) this.service.advice("default adviced", unadvicedObject );
109
110        advicedObject.length();
111    }
112
113    /**
114     * The test implements the DependentTestComponent interface therefore we
115     * are able to intercept the invocation of test(). Whereas test() invokes
116     * another advised component.
117     * @throws Exception generic exception
118     */
119    @Test
120    public void testChainedAdvices() throws Exception
121    {
122        String[] interceptorList = { LoggingInterceptorService.class.getName() };
123        this.advisedThis = (DependentTestComponent) this.service.advice(interceptorList, this);
124        this.advisedThis.test();
125    }
126
127    /**
128     * Advice a StringBuilder based on the CharSequenceInterface
129     * @throws Exception generic exception
130     */
131    @Test
132    public void testMultipleProxies() throws Exception
133    {
134        String[] interceptorList = { LoggingInterceptorService.class.getName() };
135        StringBuilder unadvicedObject = new StringBuilder("foo");
136        CharSequence advicedObject = (CharSequence) this.service.advice("first advice", interceptorList, unadvicedObject);
137        CharSequence advicedAdvicedObject = (CharSequence) this.service.advice("second advice", interceptorList, advicedObject );
138
139        assertEquals(3, advicedAdvicedObject.length());
140        assertTrue(this.service.isAdviced(advicedAdvicedObject));
141    }
142
143    /* (non-Javadoc)
144     * Advice a StringBuilder based on the CharSequenceInterface
145     * @see org.apache.fulcrum.yaafi.DependentTestComponent#test()
146     */
147    @Test
148    public void test()
149    {
150        try
151        {
152            DependentTestComponent testComponent = (DependentTestComponent) this.container.lookup(
153                DependentTestComponent.class.getName()
154                );
155
156            testComponent.test();
157        }
158        catch (ServiceException e)
159        {
160            throw new RuntimeException(e.getMessage());
161        }
162    }
