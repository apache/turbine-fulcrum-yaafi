026import java.io.File;
027import java.io.IOException;
028
029import org.apache.avalon.framework.configuration.Configuration;
030import org.apache.avalon.framework.configuration.ConfigurationUtil;
031import org.apache.avalon.framework.configuration.DefaultConfigurationBuilder;
032import org.apache.avalon.framework.context.DefaultContext;
033import org.apache.avalon.framework.service.ServiceException;
034import org.apache.fulcrum.yaafi.TestComponent;
035import org.apache.fulcrum.yaafi.framework.container.ServiceContainer;

038
039/**
040 * Test suite for the ServiceContainerFactory.
041 *
042 * @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl</a>
043 */
044
045public class ServiceContainerFactoryTest
046{
047    private ServiceContainer container = null;
048
049    @AfterEach
050    protected void tearDown() throws Exception
051    {
052        ServiceContainerFactory.dispose( this.container );
053    }
054
055    /**
056     * @throws Exception generic exception
057     */
058    private void checkTestComponent() throws Exception
059    {
060        TestComponent testComponent = this.getTestComponent();
061
062        testComponent.test();
063
064        assertEquals( testComponent.getBar(), "BAR" );
065        assertEquals( testComponent.getFoo(), "FOO" );
066
067        assertNotNull( testComponent.getUrnAvalonClassLoader() );
068        assertNotNull( testComponent.getUrnAvaloneHome() );
069        assertNotNull( testComponent.getUrnAvaloneTemp() );
070        assertNotNull( testComponent.getUrnAvalonName() );
071        assertNotNull( testComponent.getUrnAvalonPartition() );
072
073        try
074        {
075            testComponent.createException( "enforce exception", this );
076        } catch (Exception e)
077        {
078            // nothing to do
079        }
080    }
081
082    /**
083     * @return get our simple test component
084     * @throws ServiceException if service not found
085     */
086    private TestComponent getTestComponent() throws ServiceException
087    {
088        return (TestComponent) container.lookup( TestComponent.ROLE );
089    }
090
091    /**
092     * Creates a YAAFI container using a container configuration file which already contains
093     * most of the required settings
094     * 
095     * @throws Exception generic exception
096     */
097    @Test
098    public void testCreationWithContainerConfiguration() throws Exception
099    {
100        ServiceContainerConfiguration config = new ServiceContainerConfiguration();
101        config.loadContainerConfiguration( "./src/test/TestYaafiContainerConfig.xml" );
102        this.container = ServiceContainerFactory.create( config );
103        this.checkTestComponent();
104        System.out.println( this.container.toString() );
105        return;
106    }
107
108    /**
109     * Creates a YAAFI container using a non-existent container configuration file. Therefore
110     * the creation should fail.
111     * 
112     * @throws Exception generic exception
113     */
114    @Test
115    public void testCreationWithMissingContainerConfiguration() throws Exception
116    {
117        ServiceContainerConfiguration config = new ServiceContainerConfiguration();
118
119        try
120        {
121            config.loadContainerConfiguration( "./src/test/MissingTestContainerConfig.xml" );
122            this.container = ServiceContainerFactory.create( config );
123            fail( "The creation of the YAAFI container must fail" );
124        } catch (IOException e)
125        {
126            // nothing to do
127        } catch (Exception e)
128        {
129            fail( "We are expecting an IOException" );
130        }
131    }
132
133    /**
134     * Creates a YAAFI container providing all required settings manually
135     * 
136     * @throws Exception generic exception
137     */
138    @Test
139    public void testCreationWithManualSettings() throws Exception
140    {
141        ServiceContainerConfiguration config = new ServiceContainerConfiguration();
142        config.setComponentRolesLocation( "./src/test/TestRoleConfig.xml" );
143        config.setComponentConfigurationLocation( "./src/test/TestComponentConfig.xml" );
144        config.setParametersLocation( "./src/test/TestParameters.properties" );
145        this.container = ServiceContainerFactory.create( config );
146        this.checkTestComponent();
147    }
148
149    /**
150     * Creates a YAAFI container providing a Phoenix context
151     * 
152     * @throws Exception generic exception
153     */
154    @Test
155    public void testCreationWithPhoenixContext() throws Exception
156    {
157        ServiceContainerConfiguration config = new ServiceContainerConfiguration();
158        DefaultContext context = new DefaultContext();
159
160        // use an existing container configuration
161
162        config.loadContainerConfiguration( "./src/test/TestPhoenixContainerConfig.xml" );
163
164        // fill the context with Phoenix settings
165
166        context.put( "app.name", "ServiceContainerFactoryTest" );
167        context.put( "block.name", "fulcrum-yaafi" );
168        context.put( "app.home", new File( new File( "" ).getAbsolutePath() ) );
169
170        // create an instance
171
172        this.container = ServiceContainerFactory.create( config, context );
173
174        // execute the test component
175
176        this.checkTestComponent();
177    }
178
179    /**
180     * Creates a YAAFI container providing a Fortress context
181     * 
182     * @throws Exception generic exception
183     */
184    @Test
185    public void testCreationWithFortressContext() throws Exception
186    {
187        ServiceContainerConfiguration config = new ServiceContainerConfiguration();
188        DefaultContext context = new DefaultContext();
189
190        // use an existing container configuration
191
192        config.loadContainerConfiguration( "./src/test/TestFortressContainerConfig.xml" );
193
194        // fill the context with Fortress settings
195
196        context.put( "component.id", "ServiceContainerFactoryTest" );
197        context.put( "component.logger", "fulcrum-yaafi" );
198        context.put( "context-root", new File( new File( "" ).getAbsolutePath() ) );
199        context.put( "impl.workDir", new File( new File( "" ).getAbsolutePath() ) );
200
201        // create an instance
202
203        this.container = ServiceContainerFactory.create( config, context );
204
205        // execute the test component
206
207        this.checkTestComponent();
208    }
209
210    /**
211     * Reconfigures the YAAFI container with the "TestReconfigurationConfig.xml"
212     * 
213     * @throws Exception generic exception
214     */
215    @Test
216    public void testReconfiguration() throws Exception
217    {
218        // create a YAAFI instance
219
220        ServiceContainerConfiguration config = new ServiceContainerConfiguration();
221        config.loadContainerConfiguration( "./src/test/TestYaafiContainerConfig.xml" );
222        this.container = ServiceContainerFactory.create( config );
223        this.checkTestComponent();
224
225        // load a different configuration and reconfigure YAAFI
226
227        DefaultConfigurationBuilder builder = new DefaultConfigurationBuilder();
228        Configuration configuration = builder.buildFromFile( "./src/test/TestReconfigurationConfig.xml" );
229        System.out.println( ConfigurationUtil.toString( configuration ) );
230
231        this.container.reconfigure( configuration );
232        TestComponent testComponent = this.getTestComponent();
233        testComponent.test();
234
235        // the TestReconfigurationConfig.xml overwrites the
236        // TestComponentImpl.foo and the SystemProperty.FOO
237
238        assertEquals( System.getProperty( "FOO" ), "YAAFI" );
239        assertEquals( testComponent.getFoo(), "YAAFI" );
240
241    }
242}
