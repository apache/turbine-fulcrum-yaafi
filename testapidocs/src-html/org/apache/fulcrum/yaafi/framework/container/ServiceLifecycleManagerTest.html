073    /**
074     * Check our TestComponent.
075     * 
076     * @throws Exception generic exception
077     */
078    private void checkTestComponent() throws Exception
079    {
080        TestComponent testComponent = (TestComponent) container.lookup( TestComponent.ROLE );
081
082        testComponent.test();
083
084        assertEquals( testComponent.getBar(), "BAR" );
085        assertEquals( testComponent.getFoo(), "FOO" );

087        assertNotNull( testComponent.getUrnAvalonClassLoader() );
088        assertNotNull( testComponent.getUrnAvaloneHome() );
089        assertNotNull( testComponent.getUrnAvaloneTemp() );
090        assertNotNull( testComponent.getUrnAvalonName() );
091        assertNotNull( testComponent.getUrnAvalonPartition() );

094    /**
095     * Gets a list of all available services and dumps them on System.out.
096     * 
097     * @throws Exception generic exception
098     */
099    @Test
100    public void testGetServiceComponents() throws Exception
101    {
102        RoleEntry[] list = this.lifecycleManager.getRoleEntries();
103        assertNotNull( list );
104        assertTrue( list.length > 0 );
105        for (RoleEntry entry : list)
106            System.out.println( entry.toString() );
107    }
108
109    /**
110     * Reconfigure the all services
111     * 
112     * @throws Exception generic exception
113     */
114    @Test
115    public void testGeneralReconfiguration() throws Exception
116    {
117        RoleEntry[] list = this.lifecycleManager.getRoleEntries();
118
119        for (int i = 0; i < list.length; i++)
120        {
121            String serviceName = list[i].getName();
122            System.out.println( "Reconfiguring " + serviceName + " ..." );
123
124            String[] serviceNames = { list[i].getName() };
125            this.lifecycleManager.reconfigure( serviceNames );
126            assertTrue( this.container.hasService( serviceName ) );
127            assertNotNull( this.container.lookup( serviceName ) );
128        }
129    }
130
131    /**
132     * Decommission and resurrect all services
133     * 
134     * @throws Exception generic exception
135     */
136    @Test
137    public void testGeneralDecommision() throws Exception
138    {
139        String serviceName = null;
140        RoleEntry[] list = this.lifecycleManager.getRoleEntries();
141
142        for (int i = 0; i < list.length; i++)
143        {
144            serviceName = list[i].getName();
145            System.out.println( "Decommissiong " + serviceName + " ..." );
146
147            assertTrue( this.container.hasService( serviceName ) );
148            this.lifecycleManager.decommission( serviceName );
149            assertTrue( this.container.hasService( serviceName ) );
150            this.container.lookup( serviceName );
151            assertTrue( this.container.hasService( serviceName ) );
152            this.lifecycleManager.decommission( serviceName );
153            assertTrue( this.container.hasService( serviceName ) );
154        }
155    }
156
157    /**
158     * Decommission and resurrect all services
159     * 
160     * @throws Exception generic exception
161     */
162    @Test
163    public void testGeneralReconfigurationAndDecommision() throws Exception
164    {
165        String serviceName = null;
166        RoleEntry[] list = this.lifecycleManager.getRoleEntries();
167        DefaultConfigurationBuilder builder = new DefaultConfigurationBuilder();
168        Configuration configuration = builder.buildFromFile( "./src/test/TestReconfigurationConfig.xml" );
169
170        for (int i = 0; i < list.length; i++)
171        {
172            serviceName = list[i].getName();
173            String[] serviceNames = { list[i].getName() };
174            System.out.println( "Processing " + serviceName + " ..." );
175
176            // reconfigure/decommission/reconfigure the service
177            this.lifecycleManager.reconfigure( serviceNames );
178            this.lifecycleManager.decommission( serviceName );
179            this.lifecycleManager.reconfigure( serviceNames );
180
181            // run a reconfiguration over all services
182            this.container.reconfigure( configuration );
183
184            // reconfigure/decommission/reconfigure the service
185            this.container.lookup( serviceName );
186            this.lifecycleManager.reconfigure( serviceNames );
187            this.lifecycleManager.decommission( serviceName );
188            this.lifecycleManager.reconfigure( serviceNames );
189        }
190    }
191
192    /**
193     * Decommissions the TestComponent and ReconfigurationService to start them again.
194     * 
195     * @throws Exception generic exception
196     */
197    @Test
198    public void testIndividualDecommission() throws Exception
199    {
200        String serviceName = null;
201
202        // teminate the TestComponent and run it again
203
204        serviceName = TestComponent.class.getName();
205
206        this.checkTestComponent();
207        this.lifecycleManager.decommission( serviceName );
208        this.checkTestComponent();
209
210        // terminate the ReconfigurationService which is currently
211        // not instantiated and resurrect it. The moment the
212        // ReconfigurationService is instantiated it is starting to
213        // work
214
215        serviceName = ReconfigurationService.class.getName();
216
217        this.lifecycleManager.decommission( ReconfigurationService.class.getName() );
218        this.container.lookup( ReconfigurationService.class.getName() );
219
220        // now we should see that the service is starting up
221
222        Thread.sleep( 5000 );
223
224        // and terminate it again
225
226        this.lifecycleManager.decommission( ReconfigurationService.class.getName() );
227    }
