024import java.io.File;
025
026import org.apache.avalon.framework.activity.Disposable;
027import org.apache.avalon.framework.activity.Initializable;
028import org.apache.avalon.framework.configuration.Configuration;
029import org.apache.avalon.framework.configuration.ConfigurationException;
030import org.apache.avalon.framework.configuration.Reconfigurable;
031import org.apache.avalon.framework.context.Context;
032import org.apache.avalon.framework.context.ContextException;
033import org.apache.avalon.framework.context.Contextualizable;
034import org.apache.avalon.framework.logger.AbstractLogEnabled;
035import org.apache.avalon.framework.parameters.ParameterException;
036import org.apache.avalon.framework.parameters.Parameterizable;
037import org.apache.avalon.framework.parameters.Parameters;
038
039/**
040 * Implementation of the test component.
041 *
042 * @author <a href="mailto:quintonm@bellsouth.net">Quinton McCombs</a>
043 */
044public class TestComponentImpl
045        extends AbstractLogEnabled
046        implements Initializable, Reconfigurable, Parameterizable, Disposable, TestComponent, Contextualizable
047{
048    public File urnAvaloneHome;
049    public File urnAvaloneTemp;
050    public String urnAvalonPartition;
051    public String urnAvalonName;
052    public ClassLoader urnAvalonClassLoader;
053
054    public String foo;
055    public String bar;
056    public boolean decomissioned;
057    public String componentName;
058
059    /* (non-Javadoc)
060     * @see org.apache.avalon.framework.activity.Initializable#initialize()
061     */
062    public void initialize() throws Exception
063    {
064        getLogger().debug("initialize() was called");
065        decomissioned = false;
066    }
067
068    /* (non-Javadoc)
069     * @see org.apache.avalon.framework.context.Contextualizable#contextualize(org.apache.avalon.framework.context.Context)
070     */
071    public void contextualize(Context context) throws ContextException
072    {
073        this.urnAvaloneHome = (File) context.get( "urn:avalon:home" );
074        this.urnAvaloneTemp = (File) context.get( "urn:avalon:temp" );
075        this.urnAvalonName = (String) context.get( "urn:avalon:name" );
076        this.urnAvalonPartition = (String) context.get( "urn:avalon:partition" );
077        this.urnAvalonClassLoader = (ClassLoader) context.get( "urn:avalon:classloader" );
078    }
079
080    /* (non-Javadoc)
081     * @see org.apache.avalon.framework.configuration.Configurable#configure(org.apache.avalon.framework.configuration.Configuration)
082     */
083    public void configure(Configuration configuration) throws ConfigurationException
084    {
085        this.foo = configuration.getChild("FOO").getValue("FOO Not Found?!");
086    }
087
088    /* (non-Javadoc)
089     * @see org.apache.avalon.framework.configuration.Reconfigurable#reconfigure(org.apache.avalon.framework.configuration.Configuration)
090     */
091    public void reconfigure(Configuration configuration)
092        throws ConfigurationException
093    {
094        this.configure(configuration);
095    }
096
097    /* (non-Javadoc)
098     * @see org.apache.avalon.framework.parameters.Parameterizable#parameterize(org.apache.avalon.framework.parameters.Parameters)
099     */
100    public void parameterize(Parameters parameters) throws ParameterException
101    {
102        this.bar = parameters.getParameter("BAR", "BAR Not Found?!");
103    }
104
105    /* (non-Javadoc)
106     * @see org.apache.avalon.framework.activity.Disposable#dispose()
107     */
108    public void dispose()
109    {
110        getLogger().debug("dispose() was called");
111        this.decomissioned=true;
112    }
113
114    /* (non-Javadoc)
115     * @see org.apache.fulcrum.yaafi.TestComponent#test()
116     */
117    public void test()
118    {
119        setupLogger(this, "TestComponent");
120        getLogger().debug("TestComponent.test() was called");
121        getLogger().debug("urnAvaloneHome = " + this.urnAvaloneHome.toString());
122        getLogger().debug("urnAvaloneTemp = " + this.urnAvaloneTemp.toString());
123        getLogger().debug("urnAvalonPartition = " + this.urnAvalonPartition);
124        getLogger().debug("urnAvalonName = " + this.urnAvalonName);
125        getLogger().debug("foo = " + this.foo );
126        getLogger().debug("bar = " + this.bar );
127    }
128    /**
129     * @return Returns the bar.
130     */
131    public String getBar()
132    {
133        return bar;
134    }
135    /**
136     * @return Returns the componentName.
137     */
138    public String getComponentName()
139    {
140        return componentName;
141    }
142    /**
143     * @return Returns the decomissioned.
144     */
145    public boolean isDecomissioned()
146    {
147        return decomissioned;
148    }
149    /**
150     * @return Returns the foo.
151     */
152    public String getFoo()
153    {
154        return foo;
155    }
156    /**
157     * @return Returns the urnAvalonClassLoader.
158     */
159    public ClassLoader getUrnAvalonClassLoader()
160    {
161        return urnAvalonClassLoader;
162    }
163    /**
164     * @return Returns the urnAvaloneHome.
165     */
166    public File getUrnAvaloneHome()
167    {
168        return urnAvaloneHome;
169    }
170    /**
171     * @return Returns the urnAvaloneTemp.
172     */
173    public File getUrnAvaloneTemp()
174    {
175        return urnAvaloneTemp;
176    }
177    /**
178     * @return Returns the urnAvalonName.
179     */
180    public String getUrnAvalonName()
181    {
182        return urnAvalonName;
183    }
184    /**
185     * @return Returns the urnAvalonPartition.
186     */
187    public String getUrnAvalonPartition()
188    {
189        return urnAvalonPartition;
190    }
191
192    /* (non-Javadoc)
193     * @see org.apache.fulcrum.yaafi.TestComponent#createException(java.lang.String, java.lang.Object)
194     */
195    public void createException(String reason, Object caller)
196    {
197        throw new RuntimeException(reason);
198    }
199
200    /* (non-Javadoc)
201     * @see org.apache.fulcrum.yaafi.TestComponent#doSomething(long, java.lang.Object)
202     */
203    public void doSomething(long millis, Object arg)
204    {
205        try
206        {
207            Thread.sleep(millis);
208        }
209        catch (InterruptedException e)
210        {
211            // nothing to do
212        }
213    }
