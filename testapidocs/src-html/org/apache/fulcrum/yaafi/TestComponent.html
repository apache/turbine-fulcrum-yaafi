023import java.io.File;
024
025/**
026 * This is a simple component that is only used to test the avalon component
027 * service.
028 *
029 * @author <a href="mailto:quintonm@bellsouth.net">Quinton McCombs</a>
030 */
031public interface TestComponent
032{
033    /** the role name of the service */
034    static final String ROLE = "org.apache.fulcrum.yaafi.TestComponent";
035
036    /** the service method to be called during testing */
037    void test();
038
039    /**
040     * @return Returns the bar.
041     */
042    String getBar();
043
044    /**
045     * @return Returns the componentName.
046     */
047    String getComponentName();
048
049    /**
050     * @return Returns the decomissioned.
051     */
052    boolean isDecomissioned();
053
054    /**
055     * @return Returns the foo.
056     */
057    String getFoo();
058
059    /**
060     * @return Returns the urnAvalonClassLoader.
061     */
062    ClassLoader getUrnAvalonClassLoader();
063
064    /**
065     * @return Returns the urnAvaloneHome.
066     */
067    File getUrnAvaloneHome();
068
069    /**
070     * @return Returns the urnAvaloneTemp.
071     */
072    File getUrnAvaloneTemp();
073
074    /**
075     * @return Returns the urnAvalonName.
076     */
077    String getUrnAvalonName();
078
079    /**
080     * @return Returns the urnAvalonPartition.
081     */
082    String getUrnAvalonPartition();
083
084    /**
085     * Always throws an exception
086     * @param reason reason for exception
087     * @param caller the method or class calling the exception
088     */
089    void createException(String reason, Object caller);
090
091    /**
092     * Do something for the given time
093     * 
094     * @param millis time in miliseconds
095     * @param arg arguments
096     */
097    public void doSomething(long millis, Object arg);
098}
