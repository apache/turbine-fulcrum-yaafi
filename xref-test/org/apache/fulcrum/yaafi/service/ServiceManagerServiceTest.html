42  public class ServiceManagerServiceTest {
43  
44      private ServiceContainer container = null;
45  
46  
47  
48      @BeforeEach
49      protected void setUp() throws Exception
50      {
51          ServiceContainerConfiguration config = new ServiceContainerConfiguration();
52          config.loadContainerConfiguration( "./src/test/TestYaafiContainerConfig.xml" );
53          this.container = ServiceContainerFactory.create( config );
54      }
55  
56      @AfterEach
57      protected void tearDown() throws Exception
58      {
59          ServiceContainerFactory.dispose(this.container);
60      }
61  
62      /**
63       * Access the ServiceManagerService
64       * @throws Exception if service manager not found
65       */
66      @Test
67      public void testServiceManagerService() throws Exception
68      {
69          ServiceManagerService serviceManagerService = ServiceManagerServiceImpl.getInstance();
70          assertNotNull(serviceManagerService);
71  
72          assertNotNull( serviceManagerService.getAvalonLogger() );
73          assertNotNull( serviceManagerService.getContext().get("urn:avalon:home") );
74          assertNotNull( serviceManagerService.getContext().get("urn:avalon:temp") );
75  
76          // get the parameters
77          serviceManagerService.getParameters();
78  
79          // lookup the service
80          serviceManagerService = (ServiceManagerService) serviceManagerService.lookup( ServiceManagerService.class.getName() );
81          assertTrue( serviceManagerService.hasService( ServiceManagerService.class.getName() ) );
82          assertTrue( serviceManagerService.getServiceManager() instanceof ServiceManager );
83          serviceManagerService.release(serviceManagerService);
84      }
85  }
