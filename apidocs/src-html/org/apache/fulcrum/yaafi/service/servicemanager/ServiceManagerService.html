022import org.apache.avalon.framework.context.Context;
023import org.apache.avalon.framework.logger.Logger;
024import org.apache.avalon.framework.parameters.Parameters;
025import org.apache.avalon.framework.service.ServiceManager;
026
027/**
028 * Let's try to break the singleton addiction with this service. This
029 * service stores the instance of a service manager and allows access
030 * to this instance and related information such as
031 *
032 * <ul>
033 *   <li>Logger instance
034 *   <li>ServiceManager instance
035 *   <li>Context instance
036 *   <li>Parameters instance
037 * </ul>
038 *
039 *  @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl</a>
040 */
041
042public interface ServiceManagerService extends ServiceManager
043{
044    /**
045     * @return the Logger of the container
046     */
047    Logger getAvalonLogger();
048
049    /**
050     * @return the ServiceManager for the container
051     */
052    ServiceManager getServiceManager();
053
054    /**
055     * @return the Parameters for the container
056     */
057    Parameters getParameters();
058
059    /**
060     * @return the Context for the container
061     */
062    Context getContext();
063}
