024import org.apache.fulcrum.yaafi.framework.reflection.Clazz;
025
026/**
027 * Factory class to get a decrypting input stream for reading configuration
028 * files. The implementation uses dynamic class loading to make decryption an
029 * optional feature which is highly desirable when avoiding the ECCN export code
030 * problems.
031 *
032 * @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl </a>
033 */
034
035public class CryptoStreamFactory {
036        /** is the instance already initialized */
037        private static boolean isInitialized;
038
039        /** the factory to create encrypting input streams */
040        private static Object cryptoStreamFactory;
041
042        /** the name of the class to be loaded */
043        private static String className = "org.apache.fulcrum.jce.crypto.CryptoStreamFactoryImpl";
044
045        /**
046         * Create a (potentially) decrypting input stream using the default password.
047         *
048         * @param is          the input stream to be decrypted
049         * @param isEncrypted the encryption mode (true|false|auto)
050         * @return a decrypting input stream
051         * @throws Exception reading the input stream failed
052         */
053        public static InputStream getDecryptingInputStream(InputStream is, String isEncrypted) throws Exception {
054                InputStream result;
055
056                if (isEncrypted.equalsIgnoreCase("true")) {
057                        // a decrypting input stream was requested
058                        result = createDecryptingInputStream(is, "getInputStream");
059                } else if (isEncrypted.equalsIgnoreCase("auto") && hasCryptoStreamFactory()) {
060                        // no user-supplied preferences but crypto stream is available
061                        result = createDecryptingInputStream(is, "getSmartInputStream");
062                } else if (isEncrypted.equalsIgnoreCase("auto") && !hasCryptoStreamFactory()) {
063                        // no user-supplied perferences so we fall back to normal input stream
064                        result = is;
065                } else if (isEncrypted.equalsIgnoreCase("false")) {
066                        // just use normal input stream
067                        result = is;
068                } else {
069                        throw new IllegalArgumentException("Unknown decryption mode : " + isEncrypted);
070                }
071
072                return result;
073        }
074
075        /**
076         * Factory method to create a decrypting input stream.
077         *
078         * @param is                the input stream to be decrypted
079         * @param factoryMethodName the name of the factory method
080         * @return a decrypting input stream
081         * @throws Exception creating the decrypting input stream failed
082         */
083        private static InputStream createDecryptingInputStream(InputStream is, String factoryMethodName) throws Exception {
084                Class[] signature = { InputStream.class };
085                Object[] args = { is };
086                Object cryptoStreamFactory = getCryptoStreamFactory();
087
088                if (cryptoStreamFactory == null) {
089                        throw new IllegalStateException(
090                                        "No CryptoStreamFactory available - unable to create a decrypting input stream");
091                } else {
092                        return (InputStream) Clazz.invoke(cryptoStreamFactory, factoryMethodName, signature, args);
093                }
094        }
095
096        /**
097         * Factory method to create a CryptoStreamFactory.
098         * 
099         * @return a CryptoStreamFactory
100         * @throws Exception generic exception
101         */
102        private synchronized static Object getCryptoStreamFactory() throws Exception {
103                if (!isInitialized) {
104                        isInitialized = true;
105                        ClassLoader clazzLoader = CryptoStreamFactory.class.getClassLoader();
106
107                        if (Clazz.hasClazz(clazzLoader, className)) {
108                                Class[] signature = {};
109                                Object[] args = {};
110                                Class<?> clazz = Clazz.getClazz(clazzLoader, className);
111                                cryptoStreamFactory = Clazz.newInstance(clazz, signature, args);
112                        }
113                }
114
115                return cryptoStreamFactory;
116        }
117
118        /**
119         * @return true if a CryptoStreamFactory is available
120         * @throws Exception generic exception
121         */
122        private static boolean hasCryptoStreamFactory() throws Exception {
123                return getCryptoStreamFactory() != null;
124        }
125}
