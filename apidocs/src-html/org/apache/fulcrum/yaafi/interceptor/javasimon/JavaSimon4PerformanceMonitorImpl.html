022import java.lang.reflect.Method;
023
024import org.apache.fulcrum.yaafi.interceptor.util.MethodToStringBuilderImpl;
025import org.javasimon.SimonManager;
026import org.javasimon.Split;
027import org.javasimon.Stopwatch;
028
029/**
030 * Encapsulating the JAMon 2.x related API calls. JAMon 2.x allows for a much
031 * more powerful integration with Avalon services :
032 * <ul>
033 *  <li>custom ranges/units</li>
034 *  <li>exception monitoring</li>
035 *  <li>smooth web interface</li>
036 * </ul>
037 *
038 * @since 1.0.7
039 * @author <a href="mailto:siegfried.goeschl@it20one.at">Siegfried Goeschl</a>
040 */
041
042public class JavaSimon4PerformanceMonitorImpl implements JavaSimonPerformanceMonitor
043{
044    /** is monitoring enabled */
045    private boolean isActive;
046
047    /** the method currently monitored */
048    private Method method;
049
050    /** the split for this invocation */
051    private Split split;
052
053    /**
054     * Constructor.
055     *
056     * @param serviceName the service name of the service being monitored
057     * @param method the method to be monitored
058     * @param isActive is this an active monitor
059     */
060    public JavaSimon4PerformanceMonitorImpl(String serviceName, Method method, Boolean isActive)
061    {
062        this.method = method;
063        this.isActive = isActive.booleanValue();
064    }
065
066    /**
067     * Start the monitor.
068     */
069    public void start()
070    {
071        if(this.isActive)
072        {
073            String methodSignature = createMethodSignature(this.method, 0);
074            Stopwatch stopwatch = SimonManager.getStopwatch(methodSignature);
075            this.split = stopwatch.start();
076        }
077    }
078
079    /**
080     * Stop the monitor
081     */
082    public void stop()
083    {
084        if(this.isActive) 
085        {
086            this.split.stop();
087        }
088    }
089
090    /**
091     * Stop the monitor
092     */
093    public void stop(Throwable throwable)
094    {
095        if(this.isActive)
096        {
097            this.split.stop();
098        }
099    }
100
101    /**
102     * Create a method signature - JavaSimon does not look names with '#' in it.
103     */
104    private String createMethodSignature(Method method, int mode)
105    {
106        MethodToStringBuilderImpl methodToStringBuilder = new MethodToStringBuilderImpl(method, mode);
107        return methodToStringBuilder.toString().replace("#", "[]");
108    }
109}
